# Broadcasting - dimensions must be equal or one of them must be 1
arr = np.array([1, 2, 3])
print(arr + 10)

matrix = np.array([[1,2,3],[4,5,6]])
matrix + np.array([10, 20, 30])

# Broadcasting Rules:
# Rule 1: If two arrays differ in their number of dimensions, the shape of the one with fewer dimensions is padded with ones on its leading (left) side
# Rule 2: If the shape of the two arrays does not match in any dimension, the array with shape equal to 1 in that dimension is stretched to match the other shape
# Rule 3: If in any dimension the sizes disagree and neither is equal to 1, an error is raised

# LINEAR ALGEBRA CONTINUED
from numpy import linalg # (needed for linalg.iv, linalg.eig, etc)

# Inverse - A x A^-1 = I
# Inverse is the matrix "equivalent" of division. If A • x = B, then x = A^-1 • B
X = np.array([[1,2],[3,4]])
X_inv = linalg.inv(X)
print("This is inverse of matrix A: \n", X_inv)
# Determinate of X CANNOT be equal to 0, otherwise the matrix cannot be inverted
# (m x n) x (n x p) = (m x p)

# Eigenvalues and Eigenvectors - used in PCA (principal component analysis) to find the most important "directions" in data
# Definition: Av = λv
# v = eigenvector (direction)
# λ = eigenvalue (scaling factor)

# Eigenvectors: Represent the "direction"
# Eigenvalues: Represent the "magnitude" or importance
print("\n")
A = np.array([[4,2],
              [1,3]])

eigenvalues, eigenvectors = linalg.eig(A)

print("Eigenvalues:", eigenvalues)
print("Eigenvectors:\n", eigenvectors)
print("\n")

# Matrix Rank - the maximum number of linearly independent rows or columns, representing the dimension of the vector space spanned by them
# Counts non-zero rows in Row Echelon Form (REF)
A = np.array([[1,2],
              [2,4]])
rank = linalg.matrix_rank(A)
print("Matrix Rank:", rank)
